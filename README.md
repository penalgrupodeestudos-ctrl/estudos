                alt.addEventListener('click', function() { 

                    const questaoIndex = parseInt(this.getAttribute('data-questao')); 

                    const alternativaIndex = parseInt(this.getAttribute('data-alternativa')); 

                     

                    selecionarAlternativa(questaoIndex, alternativaIndex); 

                });

        function selecionarAlternativa(questaoIndex, alternativaIndex) { 

            // Registrar resposta 

            estado.respostas[questaoIndex] = alternativaIndex; 

             

            // Atualizar visual da questão 

            const questaoElement = document.getElementById(`questao-${questaoIndex}`); 

            const alternativas = questaoElement.querySelectorAll('.alternative'); 

             

            // Remover seleção anterior 

            alternativas.forEach(alt => { 

                alt.classList.remove('selected'); 

            }); 

             

            // Marcar alternativa selecionada 

            alternativas[alternativaIndex].classList.add('selected'); 

             

            // Verificar se a resposta está correta 

            const questao = estado.simuladoAtual.questoes[questaoIndex]; 

            const correta = alternativaIndex === questao.respostaCorreta; 

             

            // Atualizar estado 

            estado.resultados[questaoIndex] = correta; 

             

            // Aplicar cor de acordo com o resultado 

            if (correta) { 

                questaoElement.classList.remove('incorrect'); 

                questaoElement.classList.add('correct'); 

            } else { 

                questaoElement.classList.remove('correct'); 

                questaoElement.classList.add('incorrect'); 

                 

                // Destacar também a resposta correta 

                alternativas[questao.respostaCorreta].classList.add('correct'); 

            } 

        } 

 

        function finalizarSimulado() { 

            // Calcular resultados 

            const totalQuestoes = estado.simuladoAtual.questoes.length; 

            const respostas = Object.keys(estado.respostas).length; 

            const corretas = Object.values(estado.resultados).filter(r => r).length; 

            const incorretas = respostas - corretas; 

            const percentual = Math.round((corretas / totalQuestoes) * 100); 

             

            // Atualizar estatísticas 

            correctCount.textContent = corretas; 

            incorrectCount.textContent = incorretas; 

            scorePercent.textContent = `${percentual}%`; 

             

            // Mostrar resultados detalhados 

            detailedResults.innerHTML = ''; 

             

            estado.simuladoAtual.questoes.forEach((questao, index) => { 

                const resultado = estado.resultados[index]; 

                const respostaUsuario = estado.respostas[index]; 

                const classe = resultado ? 'correct' : 'incorrect'; 

                const icone = resultado ? 'fa-check' : 'fa-times'; 

                const texto = resultado ? 'Correta' : 'Incorreta'; 

                 

                const resultadoElement = document.createElement('div'); 

                resultadoElement.className = `question-container ${classe}`; 

                resultadoElement.innerHTML = ` 

                    <div class="question-header"> 

                        Questão ${index + 1}  

                        <span><i class="fas ${icone}"></i> ${texto}</span> 

                    </div> 

                    <div class="question-body"> 

                        <div class="question-text">${questao.enunciado}</div> 

                        <div class="alternatives"> 

                            ${questao.alternativas.map((alt, altIndex) => { 

                                let classeAlt = ''; 

                                if (altIndex === questao.respostaCorreta) { 

                                    classeAlt = 'correct'; 

                                } else if (altIndex === respostaUsuario && altIndex !== questao.respostaCorreta) { 

                                    classeAlt = 'incorrect'; 

                                } 

                                 

                                return ` 

                                    <div class="alternative ${classeAlt}"> 

                                        <div class="alternative-label">${String.fromCharCode(65 + altIndex)})</div> 

                                        <div class="alternative-text">${alt}</div> 

                                    </div> 

                                `; 

                            }).join('')} 

                        </div> 

                    </div> 

                `; 

                 

                detailedResults.appendChild(resultadoElement); 

            }); 

             

            // Mostrar resumo de resultados 

            resultsSummary.classList.remove('hidden'); 

        }

        function importarSimulado() { 

            let jsonData; 

             

            // Tentar obter dados do textarea primeiro 

            if (jsonText.value.trim()) { 

                try { 

                    jsonData = JSON.parse(jsonText.value); 

                } catch (e) { 

                    alert('Erro: JSON inválido no campo de texto.'); 

                    return; 

                } 

            } else if (jsonFile.files.length > 0) { 

                // Se não há texto, verificar se há arquivo 

                const file = jsonFile.files[0]; 

                const reader = new FileReader(); 

                 

                reader.onload = function(e) { 

                    try { 

                        jsonData = JSON.parse(e.target.result); 

                        processarImportacao(jsonData); 

                    } catch (error) { 

                        alert('Erro: Não foi possível ler o arquivo JSON.'); 

                    } 

                }; 

                 

                reader.readAsText(file); 

                return; 

            } else { 

                alert('Por favor, cole o JSON no campo de texto ou selecione um arquivo JSON.'); 

                return; 

            } 

             

            processarImportacao(jsonData); 

        } 

 

        function processarImportacao(jsonData) { 

            // Validar estrutura básica 

            if (!jsonData.titulo || !jsonData.questoes || !Array.isArray(jsonData.questoes)) { 

                alert('Erro: O JSON deve conter "titulo" e "questoes" (array).'); 

                return; 

            } 

             

            // Validar cada questão 

            for (let i = 0; i < jsonData.questoes.length; i++) { 

                const q = jsonData.questoes[i]; 

                if (!q.enunciado || !q.alternativas || !Array.isArray(q.alternativas) ||  

                    q.respostaCorreta === undefined || q.respostaCorreta === null) { 

                    alert(`Erro na questão ${i+1}: deve conter "enunciado", "alternativas" (array) e "respostaCorreta".`); 

                    return; 

                } 

                 

                if (q.respostaCorreta < 0 || q.respostaCorreta >= q.alternativas.length) { 

                    alert(`Erro na questão ${i+1}: "respostaCorreta" deve ser um índice válido (0 a ${q.alternativas.length - 1}).`); 

                    return; 

                } 

            } 

             

            // Criar novo simulado 

            const novoSimulado = { 

                id: gerarId(), 

                titulo: jsonData.titulo, 

                descricao: jsonData.descricao || '', 

                dataCriacao: new Date().toISOString().split('T')[0], 

                questoes: jsonData.questoes.map((q, index) => ({ 

                    id: index + 1, 

                    enunciado: q.enunciado, 

                    alternativas: q.alternativas, 

                    respostaCorreta: q.respostaCorreta 

                })) 

            }; 

             

            // Adicionar à lista 

            simulados.push(novoSimulado); 

            salvarDados(); 

             

            // Limpar campos 

            jsonText.value = ''; 

            jsonFile.value = ''; 

             

            // Atualizar interface 

            atualizarContadorSimulados(); 

            renderizarSimuladosAdmin(); 

             

            alert('Simulado importado com sucesso!'); 

        }

        }

        }
